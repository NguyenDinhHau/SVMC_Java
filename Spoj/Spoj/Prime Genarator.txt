#include <stdio.h>
#include <math.h>

int check_prime(int number)
{
	int i=0;
	int temp = sqrt((float)number);
	for(i=2; i<=temp; i++)
		if(number%i==0)return 1;
	return 0;
}
int main()
{
	int number_testcase, m, n;
	int i,j;
	scanf("%d", &number_testcase);
	for(i=0; i<number_testcase; i++)
	{
		scanf("%d",&m);
		scanf("%d",&n);
		if(m<2)m=2;
		for(j=m; j<=n; j++)	
		{
			if(j%2==0&&j>2)continue;
				if(check_prime(j)==0)printf("%d\n",j);
		}
		if(i<number_testcase-1)printf("\n");
	}
	return 0;
}
