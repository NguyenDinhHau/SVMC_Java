#include <stdio.h>

int visited[26][26];
int matrix[26][26];
int x_pos[4] = {-1, 1, 0, 0};
int y_pos[4] = {0, 0, 1, -1};
int count;
void dem_quan_den(int x, int y, int N)
{
	// danh dau no lai
	visited[x][y] = 1;
	count++;
	int i=0;
	// tim nhung thang xung quanh
	for(i=0; i<4; i++)
	{
		int x_temp, y_temp;
		x_temp = x+x_pos[i];
		y_temp = y+y_pos[i];
		if(x_temp>=0&&x_temp<N&&y_temp>=0&&y_temp<N&&visited[x_temp][y_temp]==0&&matrix[x_temp][y_temp]==1)
		{	
			dem_quan_den(x_temp, y_temp, N);
		}
	}
}

int main()
{
	int number_test, test, N;
	//freopen("data.txt", "r", stdin);
	scanf("%d", &number_test);
	for(test=0; test<number_test; test++)
	{
		scanf("%d", &N);
		int i,j;
		for(i=0; i<N; i++)
			for(j=0; j<N; j++)
				scanf("%d", &matrix[i][j]);
		int block = 0, max = 0;
		for(i=0; i<N; i++)
		{	for(j=0; j<N; j++)
			{
				if(matrix[i][j]==1&&visited[i][j]==0)
				{
					block++;
					count = 0;
					dem_quan_den(i, j, N);
					if(count>max)max = count;
				}
			}
		}
		for(i=0; i<N; i++)
			for(j=0; j<N; j++)visited[i][j] = 0;
		printf("Case #%d\n%d %d\n", test+1, block, max);
	}
}
